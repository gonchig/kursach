void encrypt(int msg,int exp,int n)
{
	int i;
	long long ak;
	ak=msg;

	for (i=1;i<exp;i++)
	{
		ak *= msg;
		ak %= n;
	}

	c=ak;
}


void decrypt (int d,int n,int co)
{
	int i,b;
	long long x,bx;
	ax=co;
	bx=ax;
	b=ar[1];
	
	for (i=1;b>=0;i++)
	{
		if (b==ar[i])
		{
			x=ax*ax%n;
			ax=(bx*x)%n;
			b--;
		}
		else 
		{
			x=ax*ax%n;
			ax=x;
			b--;
		}
		if (b>=ar[i])
			i--;
	}
}
